\section{Управление учетными записями}
\label{sec:accounts}

$if(accounts)$
$accounts$
$else$
\subsection{Жизненный цикл учетной записи}
\begin{enumerate}
    \item \textbf{Создание:}
    \begin{itemize}
        \item По заявке руководителя;
        \item С проверкой необходимости доступа.
    \end{itemize}
    
    \item \textbf{Изменение:}
    \begin{itemize}
        \item При смене должности;
        \item По запросу пользователя.
    \end{itemize}
    
    \item \textbf{Блокировка:}
    \begin{itemize}
        \item После 3 неудачных попыток входа;
        \item При увольнении сотрудника.
    \end{itemize}
    
    \item \textbf{Удаление:}
    \begin{itemize}
        \item Через 90 дней после увольнения;
        \item С архивированием данных.
    \end{itemize}
\end{enumerate}
$endif$
